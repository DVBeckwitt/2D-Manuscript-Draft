% Small helper to place a panel tag just above the top-left of arbitrary content.
% The label is part of the picture's bounding box so adjacent content cannot
% overlap it.
\newcommand{\paneltagged}[2]{%
  \begin{tikzpicture}
    \node[inner sep=0] (img) {#2};
    \node[
      anchor=south west,
      font=\bfseries,
      inner sep=1.5pt
    ] at ([xshift=0pt,yshift=3pt]img.north west) {#1};
  \end{tikzpicture}%
}

% Manuscript drafting helper. If the named image exists, it is used; otherwise
% a visible placeholder keeps the figure slot and caption in the draft.
\newcommand{\draftfigureplaceholder}[3][0.30\textheight]{%
  \IfFileExists{#2}{%
    \includegraphics[width=\linewidth]{#2}%
  }{%
    \fbox{%
      \begin{minipage}[c][#1][c]{0.92\linewidth}
        \centering\footnotesize
        \textbf{Draft figure placeholder}\\[0.6em]
        \texttt{\detokenize{#2}}\\[0.8em]
        #3
      \end{minipage}%
    }%
  }%
}

\newcommand{\panelgraphic}[3][]{%
  \begin{tikzpicture}
    \IfFileExists{#2}{%
      \node[inner sep=0] (img) {\includegraphics[#1]{#2}};%
    }{%
      \node[inner sep=0] (img) {%%
        \fbox{%
          \begin{minipage}[c][0.14\textheight][c]{0.82\linewidth}
            \centering\scriptsize
            \textbf{Missing draft graphic}\\[0.35em]
            \texttt{\detokenize{#2}}
          \end{minipage}%
        }%
      };%
    }%
    \node[
      anchor=north west,
      font=\bfseries\footnotesize,
      fill=white,
      fill opacity=0.85,
      text opacity=1,
      inner sep=1.5pt,
      rounded corners=1pt
    ] at ([xshift=2pt,yshift=-2pt]img.north west) {#3};
  \end{tikzpicture}%
}
